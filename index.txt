            function refreshSatPasses() {
                satelliteDB.forEach(sat => {
                    const statusEl = document.getElementById('satStatus_' + sat.id);
                    if (!statusEl) return;
                    const now = new Date();
                    const jd = getJulianDate(now);
                    const info = estimateNextPass(sat, jd, state.myLat, state.myLon);
                    if (info.inPass) {
                        statusEl.textContent = 'IN RANGE';
                        statusEl.className = 'sat-status in-range';
                    } else if (info.minutesToNext < 30) {
                        statusEl.textContent = 'In ' + Math.round(info.minutesToNext) + 'min';
                        statusEl.className = 'sat-status soon';
                    } else {
                        statusEl.textContent = '~' + Math.round(info.minutesToNext / 60) + 'h';
                        statusEl.className = 'sat-status out-range';
                    }
                });
                if (state.selectedSat) {
                    const selSat = satelliteDB.find(s => s.id === state.selectedSat);
                    if (selSat) {
                        const info = estimateNextPass(selSat, getJulianDate(new Date()), state.myLat, state.myLon);
                        document.getElementById('satStatusDot').className =
                            'status-dot ' + (info.inPass ? 'active' : (info.minutesToNext < 30 ? 'warning' :
                                'idle'));
                    }
                }
            }

            function getJulianDate(date) {
                const y = date.getUTCFullYear(),
                    m = date.getUTCMonth() + 1,
                    d = date.getUTCDate();
                const h = date.getUTCHours() + date.getUTCMinutes() / 60 + date.getUTCSeconds() / 3600;
                let Y = y,
                    M = m;
                if (M <= 2) { Y--;
                    M += 12; }
                const A = Math.floor(Y / 100);
                const B = 2 - A + Math.floor(A / 4);
                return Math.floor(365.25 * (Y + 4716)) + Math.floor(30.6001 * (M + 1)) + d + h / 24 + B - 1524.5;
            }

            function parseTLEDouble(tle, start, end) {
                const s = tle.substring(start, end).trim();
                return s ? parseFloat(s) : NaN;
            }
            function estimateNextPass(sat, jd, lat, lon) {
                const mm = parseTLEDouble(sat.tle2, 52, 63);
                const periodMin = mm ? (1440 / mm) : (sat.period || 93);
                const inclination = parseTLEDouble(sat.tle2, 8, 16) || 51.64;
                const raan = parseTLEDouble(sat.tle2, 17, 25) || 0;
                const maxLat = Math.abs(inclination) + 10;
                if (Math.abs(lat) > maxLat) return { inPass: false, minutesToNext: 9999 };
                const orbitsPerDay = mm || (1440 / periodMin);
                const phase = (jd * orbitsPerDay) % 1;
                const orbitAngle = phase * 2 * Math.PI;
                const satLat = Math.asin(Math.sin(inclination * Math.PI / 180) * Math.sin(orbitAngle)) * 180 / Math.PI;
                const earthRot = ((jd - Math.floor(jd)) * 360) % 360;
                let satLon = (raan + (phase - 0.5) * 360 - earthRot) % 360;
                if (satLon > 180) satLon -= 360;
                if (satLon < -180) satLon += 360;
                let dLon = Math.abs(satLon - lon);
                if (dLon > 180) dLon = 360 - dLon;
                const roughDist = Math.sqrt(Math.pow(satLat - lat, 2) + dLon * dLon);
                const inPass = roughDist < 25;
                let minToNext = inPass ? 0 : Math.max(5, (roughDist / 180) * periodMin);
                if (minToNext > periodMin * 3) minToNext = periodMin * (roughDist / 90);
                return { inPass, minutesToNext: Math.min(minToNext, 9999) };
            }

            // ============ TLE UPDATE ============
            async function fetchTLEUpdate() {
                const url = 'https://celestrak.org/NORAD/elements/gp.php?GROUP=amateur&FORMAT=tle';
                showToast('Fetching TLE data...');
                try {
                    const resp = await fetch(url);
                    if (!resp.ok) throw new Error('HTTP ' + resp.status);
                    const text = await resp.text();
                    const lines = text.split('\n').filter(l => l.trim() !== '');
                    const tleMap = new Map();
                    for (let i = 0; i < lines.length; i += 3) {
                        const name = lines[i].trim();
                        const tle1 = lines[i + 1].trim();
                        const tle2 = lines[i + 2].trim();
                        const noradId = parseInt(tle1.split(/\s+/)[1], 10);
                        if (!isNaN(noradId)) tleMap.set(noradId, { name, tle1, tle2 });
                    }
                    let updated = 0;
                    satelliteDB.forEach(sat => {
                        if (sat.noradId && tleMap.has(sat.noradId)) {
                            const tle = tleMap.get(sat.noradId);
                            sat.tle1 = tle.tle1;
                            sat.tle2 = tle.tle2;
                            sat.name = tle.name; // update name if changed
                            updated++;
                        }
                    });
                    refreshSatPasses();
                    showToast(`TLE updated for ${updated} satellites ✓`);
                } catch (err) {
                    showToast('TLE update failed: ' + err.message, true);
                }
            }

            function calculateGridDistance(g1, g2) {
                if (!g1 || !g2 || g1.length < 4 || g2.length < 4) return null;
                const p1 = gridToLatLon(g1),
                    p2 = gridToLatLon(g2);
                if (!p1 || !p2) return null;
                return haversine(p1.lat, p1.lon, p2.lat, p2.lon);
            }

            function gridToLatLon(grid) {
                grid = grid.toUpperCase().trim();
                if (grid.length < 4) return null;
                const flon = grid.charCodeAt(0) - 65,
                    flat = grid.charCodeAt(1) - 65;
                const slon = parseInt(grid[2]) || 0,
                    slat = parseInt(grid[3]) || 0;
                let sublon = 0,
                    sublat = 0;
                if (grid.length >= 6) {
                    sublon = grid.charCodeAt(4) - 65;
                    sublat = grid.charCodeAt(5) - 65;
                }
                const lon = (flon * 20) + (slon * 2) + (sublon * 2 / 24) - 180 + (1 / 24);
                const lat = (flat * 10) + slat + (sublat / 24) - 90 + (0.5 / 24);
                return { lat, lon };
            }

            function haversine(lat1, lon1, lat2, lon2) {
                const R = 6371;
                const dLat = (lat2 - lat1) * Math.PI / 180;
                const dLon = (lon2 - lon1) * Math.PI / 180;
                const a = Math.sin(dLat / 2) ** 2 + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math
                    .sin(dLon / 2) ** 2;
                return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
            }

            // ============ APRS PACKET ============
            function buildAPRSPacket(action, target) {
                const call = state.myCall,
                    grid = state.myGrid,
                    path = state.digipath;
                let info = '';
                switch (action) {
                    case 'beacon':
                        info = '=' + latToAPRS(state.myLat) + '/' + lonToAPRS(state.myLon) + '- >Sat QSO via ' + (
                            state.selectedSat || 'sat') + ' [' + grid + ']';
                        break;
                    case 'cq_sat':
                        info = '>CQ SAT via ' + (state.selectedSat || 'sat').toUpperCase() + ' de ' + call + ' ' +
                            grid;
                        break;
                    case 'qso_report':
                        const parts = target ? target.split(' ') : [];
                        const tcall = parts[0] || 'XXXXXX';
                        const tgrid = parts[1] || '';
                        info = ':' + tcall.toUpperCase().padEnd(9) + ':QSO via SAT ' + (state.selectedSat || '')
                            .toUpperCase() + ' | RST 59 | ' + grid;
                        if (tgrid) info += ' | UR ' + tgrid;
                        break;
                    case 'msg':
                        const mto = target || 'CQ';
                        info = ':' + mto.toUpperCase().padEnd(9) + ':Hello from ' + call +
                            ' via satellite! 73';
                        break;
                    case 'telemetry':
                        info = 'T#001,' + Math.floor(Math.random() * 255) + ',' + Math.floor(Math.random() * 100) +
                            ',' + Math.floor(Math.random() * 50) + ',0,0,00000000';
                        break;
                    default:
                        info = '>' + call + ' via OrbitAPRS';
                }
                return {
                    infoField: info,
                    sourceCall: call,
                    destCall: 'APRS',
                    digipath: path,
                    fullPacket: call + '>APRS' + (path ? ',' + path : '') + ':' + info,
                };
            }

            function latToAPRS(lat) {
                const a = Math.abs(lat);
                const d = Math.floor(a),
                    m = (a - d) * 60;
                return d.toString().padStart(2, '0') + m.toFixed(2).padStart(5, '0') + (lat >= 0 ? 'N' : 'S');
            }

            function lonToAPRS(lon) {
                const a = Math.abs(lon);
                const d = Math.floor(a),
                    m = (a - d) * 60;

            function buildAX25Frame(packet) {
                const dest = encodeAX25Address(packet.destCall || 'APRS', 0);
                const src = encodeAX25Address(packet.sourceCall, 0);
                let digi = [];
                if (packet.digipath && packet.digipath !== 'DIRECT') {
                    const dlist = packet.digipath.split(',');
                    dlist.forEach((d, i) => {
                        const dd = d.trim();
                        if (dd) digi = digi.concat(Array.from(encodeAX25Address(dd, i === dlist.length - 1 ?
                            0x01 : 0)));
                    });
                }
                const ctrl = 0x03,
                    pid = 0xF0;
                const info = stringToBytes(packet.infoField);
                let frame = [...dest, ...src, ...digi, ctrl, pid, ...info];
                const fcs = calcFCS(frame);
                frame.push(fcs & 0xFF, (fcs >> 8) & 0xFF);
                const stuffed = bitStuff(frame);
                const flag = 0x7E;
                const preamble = Array(40).fill(flag);
                const postamble = Array(5).fill(flag);
                const final = [...preamble, ...stuffed, ...postamble];
                return new Uint8Array(final);
            }

            function encodeAX25Address(call, flags) {
                const b = new Uint8Array(7);
                for (let i = 0; i < 6; i++) b[i] = (i < call.length ? call.charCodeAt(i) << 1 : 0x20 << 1);
                b[6] = 0x60 | (flags & 0x01);
                return b;
            }

            function calcFCS(data) {
                let fcs = 0xFFFF;
                for (let i = 0; i < data.length; i++) {
                    fcs ^= data[i];
                    for (let j = 0; j < 8; j++) fcs = (fcs & 1) ? (fcs >> 1) ^ 0x8408 : fcs >> 1;
                }
                return (~fcs) & 0xFFFF;
            }

            function bitStuff(bytes) {
                const bits = [];
                for (let i = 0; i < bytes.length; i++)
                    for (let b = 0; b < 8; b++) bits.push((bytes[i] >> b) & 1);
                const res = [];
                let ones = 0;
                for (const bit of bits) {
                    res.push(bit);
                    ones = bit ? ones + 1 : 0;
                    if (ones === 5) { res.push(0);
                        ones = 0; }
                }
                const out = [];
                for (let i = 0; i < res.length; i += 8) {
                    let byte = 0;
                    for (let b = 0; b < 8 && (i + b) < res.length; b++) byte |= res[i + b] << b;
                    out.push(byte);
                }
                return out;
            }

            function modulateAFSK(frameBytes) {
                const sampleRate = 44100,
                    baud = 1200;
                const mark = 1200,
                    space = 2200;
                const samplesPerBit = sampleRate / baud;
                const bits = [];
                for (let i = 0; i < frameBytes.length; i++)
                    for (let b = 0; b < 8; b++) bits.push((frameBytes[i] >> b) & 1);
                let nrzi = 1;
                const nrzBits = [];
                for (const b of bits) {
                    if (b === 0) nrzi = 1 - nrzi;
                    nrzBits.push(nrzi);
                }
                const totalSamples = Math.floor(nrzBits.length * samplesPerBit);
                const buffer = new Float32Array(totalSamples);
                let phase = 0;
                for (let i = 0; i < nrzBits.length; i++) {
                    const freq = nrzBits[i] ? mark : space;
                    const start = Math.floor(i * samplesPerBit);
                    const end = Math.floor((i + 1) * samplesPerBit);
                    for (let s = start; s < end && s < totalSamples; s++) {
                        phase += 2 * Math.PI * freq / sampleRate;
                        if (phase > 2 * Math.PI) phase -= 2 * Math.PI;
                        let amp = 0.7;
                        if (s < sampleRate * 0.002) amp *= s / (sampleRate * 0.002);
                        const rem = totalSamples - s;
                        if (rem < sampleRate * 0.004) amp *= rem / (sampleRate * 0.004);
                        buffer[s] = Math.sin(phase) * amp;
                    }
                }
                return { buffer, sampleRate, duration: totalSamples / sampleRate };
            }

            function playAudioBuffer(audio) {
                try {
                    if (!state.audioCtx) state.audioCtx = new(window.AudioContext || window.webkitAudioContext)();
                    if (state.audioCtx.state === 'suspended') state.audioCtx.resume();
                    const buf = state.audioCtx.createBuffer(1, audio.buffer.length, audio.sampleRate);
                    buf.getChannelData(0).set(audio.buffer);
                    const src = state.audioCtx.createBufferSource();
                    src.buffer = buf;
                    const gain = state.audioCtx.createGain();
                    gain.gain.value = 0.8;
                    src.connect(gain);
                    gain.connect(state.audioCtx.destination);
                    src.start();
                } catch (e) {
                    showToast('Audio error: ' + e.message, true);
                }
            }

            function genAFSKAudio(nrzBits, mark, space, samplesPerBit, fs) {
                const totalSamples = nrzBits.length * samplesPerBit;
                const buf = new Float64Array(totalSamples);
                for (let i = 0; i < nrzBits.length; i++) {
                    const freq = nrzBits[i] ? mark : space;
                    for (let s = 0; s < samplesPerBit; s++) {
                        const idx = i * samplesPerBit + s;
                        buf[idx] = Math.sin(2 * Math.PI * freq * idx / fs);
                    }
                }
                return buf;
            }

            // ============ AFSK DEMODULATOR (RX) ============
            // Dual-tone correlators (1200Hz Mark, 2200Hz Space) + DPLL bit sync
            // cross = pwr2200 - pwr1200; cross > 0  → Space, cross < 0 → Mark
            // NRZI: transition (tone changed) → decoded=0, hold → decoded=1
            class AFSKDemodulator {
                constructor() {
                    this.fs = 48000;
                    this.bitPeriod = 40;
                    this.lpfAlpha = 0.95;
                    this.dpllStep = 256;
                    this.dpllTarget = this.bitPeriod * 256;
                    this.dpllCnt = 0;
                    this.prevSign = false;
                    this.ph1200 = 0; this.ph2200 = 0;
                    this.i1200 = 0; this.q1200 = 0;
                    this.i2200 = 0; this.q2200 = 0;
                    this.prevTone = 0;
                    this.flagShift = 0;
                    this.inFrame = false;
                    this.onesCount = 0;
                    this.frameBytes = [];
                    this.bitAccum = 0;
                    this.bitPos = 0;
                    this.onPacket = null;
                    this.lastCross = 0;
                    this.lastDevNorm = 0;
                    this.lastBit = undefined;
                    this.inputGain = 1;
                    this._flagDetects = 0;
                    this._frameAttempts = 0;
                }
                configure(fs) {
                    this.fs = fs;
                    this.bitPeriod = Math.round(fs / 1200);
                    this.lpfAlpha = 0.95;
                    this.dpllTarget = this.bitPeriod * 256;
                    this.dpllCnt = 0;
                    this.prevSign = false;
                    this.ph1200 = 0; this.ph2200 = 0;
                    this.i1200 = 0; this.q1200 = 0;
                    this.i2200 = 0; this.q2200 = 0;
                    this.prevTone = 0;
                    this.flagShift = 0;
                    this.inFrame = false;
                    this.onesCount = 0;
                    this.frameBytes = [];
                    this.bitAccum = 0;
                    this.bitPos = 0;
                    this.lastCross = 0;
                    this.lastDevNorm = 0;
                    this.lastBit = undefined;
                    this.inputGain = 1;
                    this._flagDetects = 0;
                    this._frameAttempts = 0;
                }
                feed(samples) {
                    let sumSq = 0;
                    for (let i = 0; i < samples.length; i++) sumSq += samples[i] * samples[i];
                    const rms = Math.sqrt(sumSq / samples.length);
                    this.inputGain = 0.1 / Math.max(rms, 1e-6);
                    if (this.inputGain > 100) this.inputGain = 100;
                    const a = this.lpfAlpha;
                    for (let i = 0; i < samples.length; i++) {
                        const s = samples[i] * this.inputGain;
                        const c1200 = Math.cos(this.ph1200);
                        const s1200 = Math.sin(this.ph1200);
                        this.ph1200 += 2 * Math.PI * 1200 / this.fs;
                        if (this.ph1200 > 2 * Math.PI) this.ph1200 -= 2 * Math.PI;
                        this.i1200 = (1 - a) * (s * c1200) + a * this.i1200;
                        this.q1200 = (1 - a) * (s * s1200) + a * this.q1200;
                        const c2200 = Math.cos(this.ph2200);
                        const s2200 = Math.sin(this.ph2200);
                        this.ph2200 += 2 * Math.PI * 2200 / this.fs;
                        if (this.ph2200 > 2 * Math.PI) this.ph2200 -= 2 * Math.PI;
                        this.i2200 = (1 - a) * (s * c2200) + a * this.i2200;
                        this.q2200 = (1 - a) * (s * s2200) + a * this.q2200;
                        const p1200 = this.i1200 * this.i1200 + this.q1200 * this.q1200;
                        const p2200 = this.i2200 * this.i2200 + this.q2200 * this.q2200;
                        const cross = p2200 - p1200;
                        const pSum = p1200 + p2200;
                        const curSign = cross > 0;
                        if (curSign !== this.prevSign) {
                            const error = this.dpllCnt - this.dpllTarget / 2;
                            this.dpllCnt -= Math.floor(error * 0.15);
                            this.prevSign = curSign;
                        }
                        this.dpllCnt += this.dpllStep;
                        if (this.dpllCnt >= this.dpllTarget) {
                            this.dpllCnt -= this.dpllTarget;
                            this.lastCross = cross;
                            this.lastDevNorm = pSum > 1e-10 ? cross / pSum : 0;
                            const toneState = cross > 0 ? 1 : 0;
                            const decoded = (toneState === this.prevTone) ? 1 : 0;
                            this.prevTone = toneState;
                            this._outputBit(decoded);
                        }
                    }
                }
                _outputBit(decoded) {
                    this.lastBit = decoded;
                    if (this._verbose && this._dumpBits.length < this._dumpMax) this._dumpBits.push(decoded);
                    this.flagShift = ((this.flagShift << 1) | decoded) & 0xFF;
                    if (!this.inFrame) {
                        if (this.flagShift === 0x7E) {
                            this._flagDetects = (this._flagDetects || 0) + 1;
                            this.inFrame = true;
                            this.frameBytes = [];
                            this.onesCount = 0;
                            this.bitAccum = 0;
                            this.bitPos = 0;
                        }
                    } else {
                        if (decoded === 0 && this.onesCount === 5) {
                            this.onesCount = 0;
                        } else {
                            if (decoded === 1) { this.onesCount++; } else { this.onesCount = 0; }
                            this.bitAccum = (this.bitAccum >> 1) | (decoded << 7);
                            this.bitPos++;
                            if (this.bitPos === 8) {
                                if (this.bitAccum === 0x7E && this.frameBytes.length > 6) {
                                    this._decodeFrame(this.frameBytes);
                                    this.inFrame = false;
                                    this.frameBytes = [];
                                    this.onesCount = 0;
                                } else if (this.bitAccum !== 0x7E) {
                                    this.frameBytes.push(this.bitAccum);
                                }
                                this.bitPos = 0;
                                this.bitAccum = 0;
                            }
                        }
                    }
                }
                _decodeFrame(bytes) {
                    this._frameAttempts = (this._frameAttempts || 0) + 1;
                    if (bytes.length < 18) return;
                    const fcs = (bytes[bytes.length - 2] | (bytes[bytes.length - 1] << 8));
                    const data = bytes.slice(0, -2);
                    let calc = 0xFFFF;
                    for (let i = 0; i < data.length; i++) {
                        calc ^= data[i];
                        for (let j = 0; j < 8; j++) calc = (calc & 1) ? (calc >> 1) ^ 0x8408 : calc >> 1;
                    }
                    calc = (~calc) & 0xFFFF;
                    if (calc !== fcs) {
                        if (bytes.length >= 14) {
                            let dst = '', src = '';
                            for (let i = 0; i < 6; i++) dst += String.fromCharCode(bytes[i] >> 1);
                            for (let i = 7; i < 13; i++) src += String.fromCharCode(bytes[i] >> 1);
                            addTerminalLine('system', '📡 FCS fail ' + bytes.length + 'B dst=' + dst + ' src=' + src + ' fcs=0x' + fcs.toString(16) + ' calc=0x' + calc.toString(16));
                        } else {
                            this._showHex(bytes);
                        }
                        return;
                    }
                    let addrEnd = 0;
                    for (let i = 6; i < data.length; i += 7) {
                        if (data[i] & 0x01) { addrEnd = i + 1; break; }
                    }
                    if (addrEnd < 14) return;
                    let srcCall = '';
                    for (let i = 7; i < 13; i++) {
                        const c = String.fromCharCode(data[i] >> 1);
                        if (c !== ' ') srcCall += c;
                    }
                    let dstCall = '';
                    for (let i = 0; i < 6; i++) {
                        const c = String.fromCharCode(data[i] >> 1);
                        if (c !== ' ') dstCall += c;
                    }
                    let infoField = '';
                    for (let i = addrEnd + 2; i < data.length; i++) {
                        infoField += String.fromCharCode(data[i]);
                    }
                    if (this.onPacket) this.onPacket({ source: srcCall, dest: dstCall, info: infoField });
                }
                _showHex(bytes) {
                    let hex = '';
                    for (let i = 0; i < bytes.length && hex.length < 120; i++) hex += bytes[i].toString(16).padStart(2, '0') + ' ';
                    addTerminalLine('system', '📡 Raw ' + bytes.length + ' bytes: ' + hex + (bytes.length > 15 ? '...' : ''));
                }
            }

                            if (bitEl) bitEl.style.color = rms < 0.003 ? '#666' : '#00e676';
                            const cntEl = document.getElementById('vuBitCnt');
                            if (cntEl) cntEl.textContent = fps + ' fps ' + sq;
                            const fCnt = document.getElementById('vuFrameCnt');
                            if (fCnt) fCnt.textContent = '📦F:' + (demod._frameAttempts || 0);
                            const flCnt = document.getElementById('vuFlagCnt');
                            if (flCnt) flCnt.textContent = '⚑F:' + (demod._flagDetects || 0);
                            const ag = document.getElementById('vuRMS');
                            if (ag) ag.textContent = 'G:' + (demod.inputGain ? demod.inputGain.toFixed(1) : '?') + 'x';
                        }
                    };
                    source.connect(processor);
                    const muteGain = ctx.createGain();
                    muteGain.gain.value = 0;
                    processor.connect(muteGain);
                    muteGain.connect(ctx.destination);
                    state.rxSource = source;
                    state.rxProcessor = processor;
                    state.rxDemod = demod;
                    state.rxActive = true;
                    btn.className = 'btn btn-sm btn-rx active';
                    btn.innerHTML = '<span class="rx-dot"></span>RX';
                    btn.disabled = false;
                    showToast('RX started ✓');
                }).catch(err => {
                    btn.className = 'btn btn-sm btn-rx';
                    btn.innerHTML = '<span class="rx-dot"></span>RX';
                    btn.disabled = false;
                    let msg = err.message;
                    if (location.protocol === 'file:') msg = 'file:// blocked. Use a web server: npx serve .';
                    else if (err.name === 'NotAllowedError' || err.name === 'PermissionDeniedError') msg = 'Mic denied. Check browser & Windows mic permissions.';
                    else if (err.name === 'NotFoundError') msg = 'No mic found. Plug one in or select a device in Settings.';
                    showToast('RX error: ' + msg, true);
                });
            }

            function stopRX() {
                if (state.rxProcessor) {
                    try { state.rxProcessor.disconnect(); } catch (e) {}
                    state.rxProcessor = null;
                }
                if (state.rxSource) {
                    try { state.rxSource.disconnect(); } catch (e) {}
                    state.rxSource = null;
                }
                if (state.rxStream) {
                    state.rxStream.getTracks().forEach(t => t.stop());
                    state.rxStream = null;
                }
                state.rxActive = false;
                state.rxDemod = null;
                ['vuAudio','vuMark','vuSpace'].forEach(id => {
                    const el = document.getElementById(id);
                    if (el) el.style.width = '0%';
                });
                const pk = document.getElementById('vuPeak');
                if (pk) pk.textContent = '-';
                const btn = document.getElementById('rxButton');
                btn.className = 'btn btn-sm btn-rx';
                btn.innerHTML = '<span class="rx-dot"></span>RX';
                showToast('RX stopped');
            }

            function stringToBytes(str) {
                return Array.from(str).map(c => c.charCodeAt(0) & 0xFF);
            }

                    e.preventDefault();
                    if (document.getElementById('quickAction').value) buildAndSendPacket();
                }
                if (e.ctrlKey && e.key === ',') {
                    e.preventDefault();
                    toggleModal('settingsModal', true);
                }
                if (e.key === 'Escape') toggleModal('settingsModal', false);
            });
